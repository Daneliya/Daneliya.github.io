    # 1гҖҒжҹҘзңӢеҸҜз”Ёзҡ„ Redis зүҲжң¬

    1. и®ҝй—® Redis й•ңеғҸеә“ең°еқҖпјҡ https://hub.docker.com/_/redis?tab=tagsгҖӮеҸҜд»ҘйҖҡиҝҮ Sort by жҹҘзңӢе…¶д»–зүҲжң¬зҡ„ RedisпјҢй»ҳи®ӨжҳҜжңҖж–°зүҲжң¬ redis:latestгҖӮд№ҹеҸҜд»ҘеңЁдёӢжӢүеҲ—иЎЁдёӯжүҫеҲ°жғіиҰҒзҡ„зүҲжң¬гҖӮ
    2. дҪҝз”Ёз”Ё docker search redis е‘Ҫд»ӨжқҘжҹҘзңӢеҸҜз”ЁзүҲжң¬пјҡ

    # 2гҖҒеҸ–жңҖж–°зүҲзҡ„ Redis й•ңеғҸ

    иҝҷйҮҢжҲ‘д»¬жӢүеҸ–е®ҳж–№зҡ„жңҖж–°зүҲжң¬зҡ„й•ңеғҸпјҡ

    docker pull redis:latest
    1

    # 3гҖҒжҹҘзңӢжң¬ең°й•ңеғҸ

    дҪҝз”Ёд»ҘдёӢе‘Ҫд»ӨжқҘжҹҘзңӢжҳҜеҗҰе·Іе®үиЈ…дәҶ redisпјҡ

    docker images
    1

    # 4гҖҒиҝҗиЎҢе®№еҷЁ

    е®үиЈ…е®ҢжҲҗеҗҺпјҢжҲ‘д»¬еҸҜд»ҘдҪҝз”Ёд»ҘдёӢе‘Ҫд»ӨжқҘиҝҗиЎҢ redis е®№еҷЁпјҡ

    docker run -itd \
  -p 6379:6379 \
  --name redis-test \
  --requirepass "redispwd" \
  redis
    1
    2
    3
    4
    5

    еҸӮж•°иҜҙжҳҺпјҡ

    • -p 6379:6379пјҡжҳ е°„е®№еҷЁжңҚеҠЎзҡ„ 6379 з«ҜеҸЈеҲ°е®ҝдё»жңәзҡ„ 6379 з«ҜеҸЈгҖӮеӨ–йғЁеҸҜд»ҘзӣҙжҺҘйҖҡиҝҮе®ҝдё»жңәip:6379 и®ҝй—®еҲ° Redis зҡ„жңҚеҠЎгҖӮ
    • --requirepassпјҡжҢҮе®ҡеҜҶз Ғ

    # 5гҖҒе®үиЈ…жҲҗеҠҹ

    жңҖеҗҺжҲ‘д»¬еҸҜд»ҘйҖҡиҝҮ docker ps е‘Ҫд»ӨжҹҘзңӢе®№еҷЁзҡ„иҝҗиЎҢдҝЎжҒҜпјҡ

    жҺҘзқҖжҲ‘д»¬йҖҡиҝҮ redis-cli иҝһжҺҘжөӢиҜ•дҪҝз”Ё redis жңҚеҠЎгҖӮ

    docker exec -it redis-test /bin/bash
    1
